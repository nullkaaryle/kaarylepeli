1   package kaarylepeli.peli;
2   
3   import java.util.List;
4   import kaarylepeli.rakennusosat.*;
5   
6   /**
7    * Pisteenlaskija-luokka auttaa Kaarylepeli-luokkaa ylläpitämällä
8    * pistetilannetta ja niihin liittyviä.
9    */
10  public class Pisteenlaskija {
11  
12      private int pisteet;
13      private int huippupisteet;
14      private boolean ennatys;
15  
16      /**
17       * Pisteenlaskija luokan konstruktori.
18       */
19      public Pisteenlaskija() {
20          this.pisteet = 0;
21          this.huippupisteet = 0;
22          this.ennatys = false;
23      }
24  
25      /**
26       * Metodin avulla voidaan hakea tämän hetkiset pisteet.
27       *
28       * @return palauttaa pisteet kokonaislukuna
29       */
30      public int haePisteet() {
31          return this.pisteet;
32      }
33  
34      /**
35       * Metodi hakee tämän pelisession huippupisteet. Lisätty lähinnä testausta
36       * varten.
37       *
38       * @return palauttaa kokonaislukuna huippupisteet
39       */
40      public int haeHuippupisteet() {
41          return this.huippupisteet;
42      }
43  
44      /**
45       * Metodin avulla selviää tuliko päättyneessä pelissä huippupisteet. Pisteet
46       * eivät tallennu.
47       *
48       * @return palauttaa true ensimmäisellä pelikerralla ja jos saadut pisteet
49       * olivat huippupisteet
50       */
51      public boolean tuliEnnatys() {
52          return this.ennatys;
53      }
54  
55      /**
56       * Metodi päivittää pistetilanteen. Jos kaaryle on onnistunut hyppäämään
57       * puolukan yli, annetaan lisäpisteitä.
58       *
59       * @param hahmot Lista Hahmoja (esim. puolukoita)
60       * @param vauhti pelin vauhti kokonaislukuna
61       */
62      public void lisaaPisteet(List<Hahmo> hahmot, int vauhti) {
63          this.pisteet++;
64  
65          for (Hahmo hahmo : hahmot) {
66              if (hahmo.haeHahmonX() >= 50 && hahmo.haeHahmonX() <= (50 + vauhti)) {
67                  this.pisteet += 50;
68              }
69          }
70      }
71  
72      /**
73       * Metodi tarkistaa saiko pelaaja huippupisteet.
74       */
75      public void tallennaHuippupisteet() {
76          if (this.pisteet > this.huippupisteet) {
77              this.huippupisteet = pisteet;
78              this.ennatys = true;
79          }
80      }
81  
82      /**
83       * Metodin avulla saadaan alustettua pisteet ja ennätystieto, kun aloitetaan
84       * uusi peli.
85       */
86      public void nollaaPisteetJaEnnatys() {
87          this.pisteet = 0;
88          this.ennatys = false;
89      }
90  
91  }
