1   package kaarylepeli.gui;
2   
3   import java.awt.*;
4   import javax.swing.*;
5   import kaarylepeli.peli.Kaarylepeli;
6   import kaarylepeli.rakennusosat.Puolukka;
7   
8   /**
9    * Piirtoalusta, joka piirtää pelin sisällön.
10   */
11  public class Piirtaja extends JPanel implements Paivitettava {
12  
13      private Kaarylepeli kaarylepeli;
14      private int leveys;
15      private int korkeus;
16  
17      public Piirtaja(Kaarylepeli peli) {
18          this.kaarylepeli = peli;
19          this.leveys = peli.haeKentanLeveys();
20          this.korkeus = peli.haeKentanKorkeus();
21      }
22  
23      /**
24       * Kutsuu JPanel-luokalta perittyä piirtometodia. Metodissa kutsut pelin
25       * sisällön piirtämiseen.
26       *
27       * @param g grafiikka
28       */
29      @Override
30      protected void paintComponent(Graphics g) {
31          super.paintComponent(g);
32          Graphics2D graf2d = (Graphics2D) g;
33          graf2d.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
34          piirraTausta(graf2d);
35          piirraPuolukat(graf2d);
36          piirraKaaryle(graf2d);
37          piirraPisteet(graf2d);
38      }
39  
40      /**
41       * Metodi piirtää pistetilanteen pelikentän vasempaan yläkulmaan, ja lopulta
42       * Game Over -tekstin.
43       *
44       * @param g grafiikka
45       */
46      public void piirraPisteet(Graphics g) {
47          g.setColor(Color.WHITE);
48          g.setFont(new Font("Arial", 1, 20));
49          g.drawString("Pisteet: " + String.valueOf(kaarylepeli.haePisteet()), 20, 30);
50  
51          if (kaarylepeli.peliJatkuu() == false) {
52              g.setFont(new Font("Arial", 1, 50));
53              g.drawString("Peli päättyi!", leveys / 2 - 150, korkeus / 2);
54          }
55      }
56  
57      /**
58       * Metodi piirtää pelin taustan ja maana toimivan perunamuusin.
59       *
60       * @param g grafiikka
61       */
62      public void piirraTausta(Graphics g) {
63          g.setColor(Color.CYAN);
64          g.fillRect(0, 0, leveys, korkeus);
65          g.setColor(Color.YELLOW);
66          g.fillRect(0, korkeus - 50, leveys, 50);
67      }
68  
69      /**
70       * Metodi piirtää pelin puolukat.
71       *
72       * @param g grafiikka
73       */
74      public void piirraPuolukat(Graphics g) {
75          g.setColor(Color.RED);
76          for (Puolukka puolukka : this.kaarylepeli.haePuolukat()) {
77              g.fillOval(puolukka.haeHahmonX(), puolukka.haeHahmonY(), 30, 30);
78          }
79      }
80  
81      /**
82       * Metodi piirtää kaaryleen.
83       *
84       * @param g grafiikka
85       */
86      public void piirraKaaryle(Graphics g) {
87          int kaaryleenX = this.kaarylepeli.haeKaaryle().haeHahmonX();
88          int kaaryleenY = this.kaarylepeli.haeKaaryle().haeHahmonY();
89          g.setColor(Color.GREEN);
90          g.fillRect(kaaryleenX, kaaryleenY, 50, 100);
91          Rectangle oval = new Rectangle();
92      }
93  
94      /**
95       * Metodi pyytää uudelleenpiirtoa eli kentän päivitystä.
96       */
97      @Override
98      public void paivita() {
99          this.repaint();
100     }
101 
102 }
