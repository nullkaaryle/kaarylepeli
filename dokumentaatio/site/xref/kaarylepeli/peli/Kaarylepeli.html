6   import kaarylepeli.gui.Paivitettava;

9   public class Kaarylepeli extends Timer implements ActionListener {
10  
11      private int korkeus;
12      private int leveys;
13      private boolean peliJatkuu;
14      private Kaaryle kaaryle;
15      private ArrayList<Puolukka> puolukat;
16      private Paivitettava paivitettava;
17  
18      public Kaarylepeli() {
19          super(1000, null);
20          this.leveys = 1000;
21          this.korkeus = 300;
22          this.peliJatkuu = true;
23          addActionListener(this);
24          setInitialDelay(2000);  //voisi myös käynnistyä, kun pelaaja painaa painiketta
25          this.kaaryle = new Kaaryle();
26          this.puolukat = luoPuolukat();
27      }
28  
29      public Kaaryle haeKaaryle() {
30          return this.kaaryle;
31      }
32  
33      public ArrayList<Puolukka> haePuolukat() {
34          return this.puolukat;
35      }
36  
37      public boolean peliJatkuu() {
38          return this.peliJatkuu;
39      }
40  
41      public int haeKentanKorkeus() {
42          return this.korkeus;
43      }
44  
45      public int haeKentanLeveys() {
46          return this.leveys;
47      }
48  
49      //-----------------------------------------------------
50      //pelisilmukan hahmotelma
51      //tällä hetkellä kentällä vain kaksi liikkuvaa objektia
52      //kaaryle ja puolukka
53      //eli näyttää siltä kuin puolukka liikkuisi kohti kaaryletta
54      //koska kentässä ei ole vielä rullaavaa taustaa
55      @Override
56      public void actionPerformed(ActionEvent painallus) {
57  
58          if (peliJatkuu == false) {
59              System.out.println("*** Peli loppui! ***");
60              //pistenäyttö + sammutus
61              return;
62          }
63  
64          kaaryleenHyppy();
65          puolukatVyoryvat();
66          tarkistaOsumat();
67          this.paivitettava.paivita();
68          this.setDelay(50);
69      }
70  
71      //------------------------------------------------------------------
72      //tällä hetkellä yksi puolukka tapetaan ja luodaan uusi Puolukka-olio
73      //jatkossa todennäköisesti useita puolukaita, joita kierrätetään kentän läpi
74      //liikkuminen aina oikealta vasemmalle
75      //10 yksikköä kerrallaan
76      public void puolukatVyoryvat() {
77          boolean puolukkaKuoli = false;
78  
79          for (Puolukka puolukka : this.puolukat) {
80              for (int i = 0; i < 10; i++) {
81                  puolukka.liikuVasemmalle();;
82              }
83  
84              if (puolukka.haeHahmonX() <= -50) {
85                  puolukkaKuoli = true;
86              }
87          }
88  
89          if (puolukkaKuoli == true) {
90              this.puolukat.remove(0);
91              this.puolukat.add(new Puolukka());
92          }
93      }
94  
95      //----------------------------------------------------
96      //yksikin puolukka on tappava
97      //tällä hetkellä puolukat ovat todellisuudessa neliöitä
98      //mutta jatkossa voidaan harkita oikeasti pyöreitä kulmia
99      public void tarkistaOsumat() {
100         for (Puolukka puolukka : this.puolukat) {
101 
102             if (this.kaaryle.osuu(puolukka)) {
103                 this.peliJatkuu = false;
104             }
105         }
106     }
107 
108     //-------------------------------------------------------
109     //yhden silmukan aikana jos hyppyarvo ei ole nolla
110     //liikutaan 10 yksikköä y-akselilla
111     //hyppyarvo pitää kirjaa ollaanko menty jo lakipisteen 100 ohi
112     //eli pitääkö liikkua jo alaspäin, ettei mennä avaruuteen
113     //jatkossa hahmolla voi olla suunta ja vauhti
114     //ja logiikka voidaan eriyttää tästä luokasta
115     public void kaaryleenHyppy() {
116         int hyppyarvo = this.kaaryle.hyppyarvo();
117 
118         if (hyppyarvo > 0 && hyppyarvo < 101) {
119             for (int i = 0; i < 10; i++) {
120                 this.kaaryle.liikuYlos();
121             }
122             this.kaaryle.kasvataHyppyarvoa(10);
123         }
124 
125         if (hyppyarvo >= 101) {
126             for (int i = 0; i < 10; i++) {
127                 this.kaaryle.liikuAlas();
128             }
129             this.kaaryle.kasvataHyppyarvoa(10);
130         }
131 
132     }
133 
134     public void asetaPaivitettava(Paivitettava paivitettava) {
135         this.paivitettava = paivitettava;
136     }
137 
138     public ArrayList<Puolukka> luoPuolukat() {
139         ArrayList<Puolukka> puolukkalista = new ArrayList<>();
140         //jatkossa arvotaan alkupuolukat ja sijainnit jonkin logiikan mukaan
141         //eivät saa olla liian lähellä toisiaan tai liian harvassa
142         puolukkalista.add(new Puolukka());
143         return puolukkalista;
144     }
