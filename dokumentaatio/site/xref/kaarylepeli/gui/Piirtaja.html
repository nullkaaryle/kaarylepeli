3   import java.awt.Color;
4   import java.awt.Graphics;
5   import java.awt.Graphics2D;
6   import javax.swing.*;
7   import kaarylepeli.peli.Kaarylepeli;
8   import kaarylepeli.rakennusosat.Puolukka;

10  public class Piirtaja extends JPanel implements Paivitettava {
11  
12      private Kaarylepeli kaarylepeli;
13  
14      public Piirtaja(Kaarylepeli peli) {
15          this.kaarylepeli = peli;
16      }
17  
18      @Override
19      protected void paintComponent(Graphics grafiikka) {
20          super.paintComponent(grafiikka);
21  
22          Graphics2D graf2d = (Graphics2D) grafiikka;
23          piirraPuolukat(graf2d);
24          piirraKaaryle(graf2d);
25      }
26  
27      public void piirraPuolukat(Graphics grafiikka) {
28          grafiikka.setColor(Color.RED);
29          for (Puolukka puolukka : this.kaarylepeli.haePuolukat()) {
30              grafiikka.fillOval(puolukka.haeHahmonX(), puolukka.haeHahmonY(), 30, 30);  //puolukan X ja y
31  
32          }
33      }
34  
35      public void piirraKaaryle(Graphics grafiikka) {
36          grafiikka.setColor(Color.GREEN);
37          grafiikka.fillRect(this.kaarylepeli.haeKaaryle().haeHahmonX(), this.kaarylepeli.haeKaaryle().haeHahmonY(), 50, 100);  //kaaryleen X ja y
38      }
39  
40      @Override
41      public void paivita() {
42          this.repaint();
43      }
44  
45  }
